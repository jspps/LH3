<form action="${2}" id="fm_page_req" method="post">
	<input type="hidden" value="${3}" id="inp_fm_page" name="inp_fm_page">
	<input type="hidden" value="${4}" id="inp_fm_pageMax" name="inp_fm_pageMax">
</form>

<script type="text/javascript">
	function click2sendPaging(curPage){
		$("#inp_fm_page").val(curPage);
		$("#fm_page_req").submit();
	};
	function click2PrePage(){
		var curPage = $("#inp_fm_page").val();
		curPage--;
		if(curPage <= 0)
			curPage = 1;
		$("#inp_fm_page").val(curPage);
		$("#fm_page_req").submit();
	};
	
	function click2NxtPage(){
		var curPage = $("#inp_fm_page").val();
		curPage++;
		var pageMax = $("#inp_fm_pageMax").val();
		if(curPage > pageMax)
			curPage = pageMax;
		$("#inp_fm_page").val(curPage);
		$("#fm_page_req").submit();
	};
</script>
